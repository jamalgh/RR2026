0.1 Overview

This report analyzes the viewership trends of Suits, an American legal drama television series that aired on USA Network from 2011 to 2019. The analysis includes data extraction, visualization, and statistical examination of season-by-season viewership patterns.

0.2 Data Collection and Setup

0.2.1 Creating the Viewership Dataset

First, let’s create the viewership data from Wikipedia sources:

```{python} import pandas as pd import matplotlib.pyplot as plt import seaborn as sns import numpy as np from pathlib import Path

1 Create data directory if it doesn’t exist

Path(‘data’).mkdir(exist_ok=True) Path(‘images’).mkdir(exist_ok=True)

2 Create viewership dataframe

viewership_data = { ‘season’: [1, 2, 3, 4, 5, 6, 7, 8, 9], ‘episodes’: [12, 16, 16, 16, 16, 16, 16, 16, 10], ‘first_aired’: [‘2011-06-23’, ‘2012-06-14’, ‘2013-07-16’, ‘2014-06-11’, ‘2015-06-24’, ‘2016-07-13’, ‘2017-07-12’, ‘2018-07-18’, ‘2019-07-17’], ‘last_aired’: [‘2011-09-08’, ‘2013-02-21’, ‘2014-04-10’, ‘2015-03-04’, ‘2016-03-02’, ‘2017-03-01’, ‘2018-04-25’, ‘2019-02-27’, ‘2019-09-25’], ‘avg_viewers_millions’: [4.16, 3.60, 2.73, 2.26, 2.01, 1.60, 1.30, 1.02, 0.99] }

df = pd.DataFrame(viewership_data)

3 Save to CSV

df.to_csv(‘data/viewership.csv’, index=False)

4 Display the data

print(“=== SUITS VIEWERSHIP DATA ===”) print(df.to_string(index=False))

5 Calculate changes

df[‘viewers_change’] = df[‘avg_viewers_millions’].diff() df[‘viewers_change_percent’] = df[‘avg_viewers_millions’].pct_change() * 100

6 Calculate key statistics

season_1_viewers = df.loc[0, ‘avg_viewers_millions’] final_season_viewers = df.loc[len(df)-1, ‘avg_viewers_millions’] total_change = final_season_viewers - season_1_viewers percent_change = (total_change / season_1_viewers) * 100

peak_season = df.loc[df[‘avg_viewers_millions’].idxmax()] lowest_season = df.loc[df[‘avg_viewers_millions’].idxmin()]

7 Display statistics

print(“=== KEY STATISTICS ===”) print(f”Season 1 Average: {season_1_viewers:.2f} million viewers”) print(f”Final Season Average: {final_season_viewers:.2f} million viewers”) print(f”Total Change: {total_change:.2f} million ({percent_change:.1f}%)“) print(f”Peak Season: Season {int(peak_season[‘season’])} with {peak_season[‘avg_viewers_millions’]:.2f}M viewers”) print(f”Lowest Season: Season {int(lowest_season[‘season’])} with {lowest_season[‘avg_viewers_millions’]:.2f}M viewers”) print(f”Episodes: {df[‘episodes’].sum()}“) print(f”Total Seasons: {len(df)}“) print(f”Series Run: {df[‘first_aired’].iloc[0]} to {df[‘last_aired’].iloc[-1]}“)

8 Create summary table

summary_data = { ‘Metric’: [‘Total Seasons’, ‘Total Episodes’, ‘Original Network’, ‘Season 1 Average’, ‘Season 9 Average’, ‘Peak Season’, ‘Lowest Season’, ‘Overall Change’, ‘Percentage Decline’, ‘Series Run’], ‘Value’: [str(len(df)), str(df[‘episodes’].sum()), ‘USA Network’, f”{season_1_viewers:.2f} million viewers”, f”{final_season_viewers:.2f} million viewers”, f”Season {int(peak_season[‘season’])} ({peak_season[‘avg_viewers_millions’]:.2f}M)“, f”Season {int(lowest_season[‘season’])} ({lowest_season[‘avg_viewers_millions’]:.2f}M)“, f”{total_change:.2f} million viewers”, f”{percent_change:.1f}%“, f”{df[‘first_aired’].iloc[0]} – {df[‘last_aired’].iloc[-1]}“] }

summary_df = pd.DataFrame(summary_data) summary_df

9 Set style

plt.style.use(‘seaborn-v0_8-whitegrid’) sns.set_palette(“husl”)

10 GRAPH 1: Viewership Over Time

fig1, ax1 = plt.subplots(figsize=(12, 6)) ax1.plot(df[‘season’], df[‘avg_viewers_millions’], marker=‘o’, linewidth=2.5, markersize=8, color=‘#2E86AB’, markerfacecolor=‘#A23B72’, markeredgewidth=2) ax1.fill_between(df[‘season’], df[‘avg_viewers_millions’], alpha=0.3, color=‘#2E86AB’)

11 Add value annotations

for i, row in df.iterrows(): ax1.annotate(f”{row[‘avg_viewers_millions’]:.2f}M”, (row[‘season’], row[‘avg_viewers_millions’]), textcoords=“offset points”, xytext=(0,10), ha=‘center’, fontsize=9)

ax1.set_title(‘Suits: Average Viewership by Season (2011-2019)’, fontsize=14, fontweight=‘bold’, pad=20) ax1.set_xlabel(‘Season’, fontsize=12) ax1.set_ylabel(‘Average Viewers (millions)’, fontsize=12) ax1.set_xticks(range(1, 10)) ax1.set_ylim(0, 5) ax1.grid(True, alpha=0.3) plt.tight_layout() plt.savefig(‘images/viewership_over_time.png’, dpi=300, bbox_inches=‘tight’) plt.show()

12 GRAPH 2: Season-to-Season Changes

fig2, ax2 = plt.subplots(figsize=(12, 6)) colors = [‘#E94F37’ if x < 0 else ‘#44AF69’ for x in df[‘viewers_change’]] bars = ax2.bar(df[‘season’], df[‘viewers_change’], color=colors, alpha=0.7, edgecolor=‘black’, linewidth=1.2)

ax2.axhline(y=0, color=‘black’, linestyle=‘-’, linewidth=1.5) ax2.set_title(‘Suits: Season-to-Season Viewership Change’, fontsize=14, fontweight=‘bold’, pad=20) ax2.set_xlabel(‘Season’, fontsize=12) ax2.set_ylabel(‘Change in Viewers (millions)’, fontsize=12) ax2.set_xticks(range(1, 10)) ax2.grid(True, alpha=0.3, axis=‘y’)

13 Add value labels on bars

for i, (season, change) in enumerate(zip(df[‘season’], df[‘viewers_change’])): if not pd.isna(change): ax2.text(season, change + (0.05 if change > 0 else -0.15), f’{change:+.2f}‘, ha=’center’, va=‘bottom’ if change > 0 else ‘top’, fontsize=9, fontweight=‘bold’)

plt.tight_layout() plt.savefig(‘images/viewership_changes.png’, dpi=300, bbox_inches=‘tight’) plt.show()

14 GRAPH 3: Percentage Change

fig3, ax3 = plt.subplots(figsize=(12, 6)) colors = [‘#E94F37’ if x < 0 else ‘#44AF69’ for x in df[‘viewers_change_percent’]] bars = ax3.bar(df[‘season’], df[‘viewers_change_percent’], color=colors, alpha=0.7, edgecolor=‘black’, linewidth=1.2)

ax3.axhline(y=0, color=‘black’, linestyle=‘-’, linewidth=1.5) ax3.set_title(‘Suits: Season-to-Season Percentage Change’, fontsize=14, fontweight=‘bold’, pad=20) ax3.set_xlabel(‘Season’, fontsize=12) ax3.set_ylabel(‘Percentage Change (%)’, fontsize=12) ax3.set_xticks(range(1, 10)) ax3.grid(True, alpha=0.3, axis=‘y’)

15 Add value labels

for i, (season, pct_change) in enumerate(zip(df[‘season’], df[‘viewers_change_percent’])): if not pd.isna(pct_change): ax3.text(season, pct_change + (2 if pct_change > 0 else -8), f’{pct_change:+.1f}%‘, ha=’center’, va=‘bottom’ if pct_change > 0 else ‘top’, fontsize=9, fontweight=‘bold’)

plt.tight_layout() plt.savefig(‘images/percentage_changes.png’, dpi=300, bbox_inches=‘tight’) plt.show()

16 Create detailed statistics table

stats_data = { ‘Statistic’: [‘Total viewership loss’, ‘Percentage decline’, ‘Average seasonal decline’, ‘Largest single-season drop’, ‘Smallest decline’, ‘Number of seasons with decline’, ‘Seasons above 3M viewers’, ‘Seasons below 1.5M viewers’], ‘Value’: [f”{abs(total_change):.2f} million viewers”, f”{abs(percent_change):.1f}%“, f”{abs(total_change)/(len(df)-1):.2f} million viewers per season”, f”{abs(df[‘viewers_change’].min()):.2f}M viewers (Season 2→3)“, f”{abs(df[‘viewers_change’].iloc[-1]):.2f}M viewers (Season 8→9)“, f”{len(df)-1} out of {len(df)-1} transitions”, f”{len(df[df[‘avg_viewers_millions’] > 3.0])} seasons”, f”{len(df[df[‘avg_viewers_millions’] < 1.5])} seasons”] }

stats_df = pd.DataFrame(stats_data) stats_df
